##### 数组嵌套

1. c++

   ```c++
   public:
       int arrayNesting(vector<int>& nums) {
           int res = 0;
           for(int i = 0;i < nums.size();i++){
               int count = 0;
               int k = i;
               while(nums[k]!=-1){
                   int tmp = k;
                   k = nums[k];
                   nums[tmp] = -1;
                   count += 1;
               }
   
               if(res<count){
                   res = count;
               }
           }
           return res;
       }
   };
   ```
